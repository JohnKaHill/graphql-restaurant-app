    filteredOrders (filterOption) {
      if (filterOption === "Active") {
        return orders.filter(function(item) {
          return item.isOpen
        });
      } else if(filterOption === "Closed") {
        return orders.filter(function(item) {
          return !item.isOpen
        });
      } else {
        return orders
